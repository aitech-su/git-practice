# 顯示當前使用者的所有進程
ps  # 預設僅顯示當前終端下的進程

# 查看當前使用者的所有進程（詳細格式）
ps -f  # 使用 -f 參數顯示更詳細的進程資訊，包括父進程 ID、啟動時間等

# 查看所有使用者的所有進程
ps -e  # 使用 -e 顯示系統中所有使用者的所有進程
# 或者
ps -A  # -A 參數與 -e 效果相同

# 查看所有進程的詳細資訊
ps -ef  # 使用 -ef 組合參數查看所有進程的詳細資訊

# 查看特定使用者的所有進程
ps -u username  # 將 username 替換為目標使用者名稱，列出該使用者的所有進程

# 查看特定進程的詳細資訊
ps -p PID -f  # 將 PID 替換為目標進程 ID，使用 -p 指定進程 ID

# 使用 ps 與 grep 查找包含特定名稱的進程
ps -ef | grep process_name  # 通過 grep 搜索指定名稱的進程

# 查看進程的層次結構
ps -ef --forest  # 使用 --forest 參數查看進程的層次結構，以樹狀顯示

# 按 CPU 或記憶體使用率排序顯示
ps aux --sort=-%cpu  # 使用 CPU 占用率從高到低排序顯示
ps aux --sort=-%mem  # 使用記憶體占用率從高到低排序顯示

# 動態顯示進程資訊（使用 top 指令）
top  # top 指令動態更新顯示進程資訊，便於實時監控
